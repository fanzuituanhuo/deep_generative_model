\section{二阶收缩映射的动机与定义}
在流形优化中，我们用\textbf{收缩映射（Retraction）}$R$ 将切空间中的步长 $s \in T_x\cM$ 映射回流形上：
\[
x^+ = R_x(s).
\]
因此，理解\textbf{拉回映射（Pullback）}
\[
\hat f_x : T_x\cM \to \R,\qquad \hat f_x(s) = (f\circ R_x)(s)
\]
在 $s=0$ 附近的二阶结构，是构造/分析二阶算法, 比如Newton、Trust-region 等的关键。

\medskip
令 $v\in T_x\cM$，考虑\textbf{收缩曲线(Retraction Curve)}
\[
c(t) = R_x(tv), \qquad c(0)=x,\ \ c'(0)=v.
\]
由上一节沿曲线的二阶泰勒展开式直接得到：
\begin{align*}
f(R_x(tv))
= &~f(x)
+ t\langle \grad f(x), v\rangle_x
+ \frac{t^2}{2}\langle \Hess f(x)[v], v\rangle_x
\\&~+ \frac{t^2}{2}\langle \grad f(x), c''(0)\rangle_x
+ O(t^3).
\end{align*}
其中 $c''(0)=\frac{D}{dt}c'(0)$ 是曲线 $c$ 在 $t=0$ 的\textbf{内蕴加速度(Intrinsic Acceleration)}。

\medskip
注意最后一项 $\frac{t^2}{2}\langle \grad f(x), c''(0)\rangle_x$ 是不希望出现的，因为它依赖于所选收缩映射 $R$,而不是仅依赖于流形几何与 $f$ 的二阶信息）。

该项在两类情形会消失：
\[
\text{(i) }\grad f(x)=0\quad\text{或}\quad \text{(ii) }c''(0)=0.
\]
前者对应“临界点附近”的分析,后者提示我们可以对收缩映射提出更强的几何要求，从而在任意点消除该项。

\begin{definition}{二阶收缩映射（second-order retraction）}{5.42}
令 $\cM$ 是黎曼流形。称收缩映射 $R$ 为\textbf{二阶收缩映射（second-order retraction）}，若对任意 $x\in \cM$ 与任意 $v\in T_x\cM$，收缩曲线
\[
c(t)=R_x(tv)
\]
在 $t=0$ 处满足的内蕴加速度为0：
\[
c''(0)=0.
\]
\end{definition}

\section{一些例子和性质}
\begin{example}{球面上的归一化收缩映射是二阶的}{5.43}
在单位球面 $S^{d-1}$ 上考虑常用收缩映射
\[
R_x(v)=\frac{x+v}{\|x+v\|}.
\]
令 $c(t)=R_x(tv)$。计算可得其（欧几里得）二阶导满足 $$\ddot c(0)=-\|v\|^2 x,$$ 它与 $T_xS^{d-1}$ 正交。
因此内蕴加速度是切向投影：
\[
c''(0)=\Proj_x(\ddot c(0))=\Proj_x(-\|v\|^2 x)=0.
\]
故该收缩映射是二阶收缩映射。
\end{example}

\begin{proposition}{两种重要特例}{5.44}
设 $\cM$ 是黎曼流形，$R$ 是任意收缩映射，$f:\cM\to \R$ 光滑，$\hat f_x=f\circ R_x$.
\begin{enumerate}
\item[(1)] \textbf{临界点处（任意收缩映射都能用）}：若 $x$ 是临界点（$\grad f(x)=0$），则
\begin{align*}
f(R_x(s)) = f(x) + \frac{1}{2}\langle \Hess f(x)[s], s\rangle_x + O(\|s\|_x^3).
\end{align*}
\item[(2)] \textbf{二阶收缩映射}：若 $R$ 是二阶收缩映射，则对任意 $x\in\cM$ 都有
\begin{align*}
\hspace{-12.3mm}f(R_x(s))
= f(x) + \langle \grad f(x), s\rangle_x
+ \frac{1}{2}\langle \Hess f(x)[s], s\rangle_x
+ O(\|s\|_x^3).
\end{align*}
\end{enumerate}
\end{proposition}

\begin{proof}
将上一节的收缩曲线展开式写成 $s=t v$ 的形式即可。
\begin{itemize}
\item 若 $\grad f(x)=0$，则加速度项自动消失，得到结论 (1)。
\item 若 $R$ 二阶，则 $c''(0)=0$，从而加速度项消失，得到结论 (2)。
\end{itemize}
并将 $O(t^3)$ 重写为 $O(\|s\|_x^3)$（因 $s=t v$ 且局部等价）。
\end{proof}

\medskip
命题~\ref{prop:5.44} 说明\textbf{在临界点附近}，即便只用一阶收缩映射，二阶展开也不会被加速度项影响；
而若我们拥有\textbf{二阶收缩映射}，则\textbf{在任意点}都可以直接把 $\hat f_x$ 当作外在空间的函数来做标准二阶近似。

\begin{proposition}{用 pullback 在切空间中计算黎曼 Hessian}{5.45}
若 $R$ 是二阶收缩映射，或 $x$ 是临界点（$\grad f(x)=0$），则
\[
\Hess f(x) \;=\; \Hess(f\circ R_x)(0),
\]
右侧是定义在欧式空间 $T_x\cM$ 上的\textbf{经典 Hessian}。
\end{proposition}

\begin{proof}
记 $\hat f_x(s)=f(R_x(s))$。
\begin{itemize}
\item 若 $R$ 二阶，由命题~\ref{prop:5.44}(2)：
\[
\hat f_x(s)= f(x)+\langle \grad f(x), s\rangle_x + \frac12\langle \Hess f(x)[s],s\rangle_x + O(\|s\|_x^3).
\]
在欧式空间 $T_x\cM$ 对 $s$ 求导（并用 $\Hess f(x)$ 自伴随性）：
\[
\grad \hat f_x(s) = \grad f(x) + \Hess f(x)[s] + O(\|s\|_x^2),
\]
\[
\Hess \hat f_x(s)[\dot s] = \Hess f(x)[\dot s] + O(\|s\|_x\|\dot s\|_x).
\]
令 $s=0$，得 $\Hess \hat f_x(0)=\Hess f(x)$。
\item 若 $\grad f(x)=0$，同理由命题~\ref{prop:5.44}(1) 出发即可。
\end{itemize}
\end{proof}

\begin{remark}{临界点处 Hessian 对度量的不变性}{}
临界点与（一阶）收缩映射的定义不依赖于黎曼度量。
命题~\ref{prop:5.45} 进一步表明：在临界点 $x$ 处，$\Hess f(x)$ 对所选黎曼度量的依赖只通过切空间上的内积 $\langle\cdot,\cdot\rangle_x$ 体现。
特别地，$\Hess f(x)$ 的符号结构（signature）（正/零/负特征值的个数）与我们选取的黎曼度量无关。
\end{remark}

\begin{remark}{二阶收缩映射的加速度保持性质, Exercise 5.46}{}
固定 $x\in\cM$，取切空间曲线 $w:I\to T_x\cM$，满足 $w(0)=0$，并诱导流形曲线
\[
c(t)=R_x(w(t)).
\]
显然有 $c(0)=x$ 且 $c'(0)=w'(0)$。若 $R$ 为二阶收缩映射，则还有
\[
c''(0)=w''(0).
\]
\textbf{一个直接推论}：给定任意 $u,v\in T_x\cM$，可以构造曲线
\[
c(t)=R_x\!\left(tu+\frac{t^2}{2}v\right)
\]
使得 $c'(0)=u$ 且 $c''(0)=v$。这从侧面解释了二阶收缩映射为何能对齐切空间中的二阶信息。
\end{remark}
